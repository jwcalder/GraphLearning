113class spectral(clustering):
114    def __init__(self, W, num_clusters, method='NgJordanWeiss', extra_dim=0):
115        """Spectral clustering
116        ===================
117
118        Implements several methods for spectral clustering, including Shi-Malik and Ng-Jordan-Weiss. See
119        the tutorial paper [1] for details.
120
121        Parameters
122        ----------
123        W : numpy array, scipy sparse matrix, or graphlearning graph object
124            Weight matrix representing the graph.
125        num_clusters : int
126            Number of desired clusters.
127        method : {'combinatorial', 'ShiMalik', 'NgJordanWeiss'} (optional), default='NgJordanWeiss'
128            Spectral clustering method.
129        extra_dim : int (optional), default=0
130            Extra dimensions to include in spectral embedding.
131        
132        Examples
133        ----
134        Spectral clustering on the two-moons dataset: [spectral_twomoons.py](https://github.com/jwcalder/GraphLearning/blob/master/examples/spectral_twomoons.py).
135        ```py
136        import numpy as np
137        import graphlearning as gl
138        import matplotlib.pyplot as plt
139        import sklearn.datasets as datasets
140
141        X,labels = datasets.make_moons(n_samples=500,noise=0.1)
142        W = gl.weightmatrix.knn(X,10)
143
144        model = gl.clustering.spectral(W, num_clusters=2)
145        pred_labels = model.fit_predict()
146
147        accuracy = gl.clustering.clustering_accuracy(pred_labels, labels)
148        print('Clustering Accuracy: %.2f%%'%accuracy)
149
150        plt.scatter(X[:,0],X[:,1], c=pred_labels)
151        plt.axis('off')
152        plt.show()
153        ```
154        Spectral clustering on MNIST: [spectral_mnist.py](https://github.com/jwcalder/GraphLearning/blob/master/examples/spectral_mnist.py).
155        ```py
156        import graphlearning as gl
157
158        W = gl.weightmatrix.knn('mnist', 10, metric='vae')
159        labels = gl.datasets.load('mnist', labels_only=True)
160
161        model = gl.clustering.spectral(W, num_clusters=10, extra_dim=4)
162        pred_labels = model.fit_predict(all_labels=labels)
163        
164        accuracy = gl.clustering.clustering_accuracy(pred_labels,labels)
165        print('Clustering Accuracy: %.2f%%'%accuracy)
166        ```
167
168        Reference
169        ---------
170        [1] U. Von Luxburg.  [A tutorial on spectral clustering.](https://link.springer.com/content/pdf/10.1007/s11222-007-9033-z.pdf) Statistics and computing 17.4 (2007): 395-416.
171        """
172        super().__init__(W, num_clusters)
173            
174        self.method = method
175        self.extra_dim = extra_dim
176
177    def _fit(self, all_labels=None):
178
179        n = self.graph.num_nodes
180        num_clusters = self.num_clusters
181        method = self.method
182        extra_dim = self.extra_dim
183
184        if method == 'combinatorial':
185            vals, vec = self.graph.eigen_decomp(k=num_clusters+extra_dim)
186        elif method == 'ShiMalik':
187            vals, vec = self.graph.eigen_decomp(normalization='randomwalk', k=num_clusters+extra_dim)
188        elif method == 'NgJordanWeiss':
189            vals, vec = self.graph.eigen_decomp(normalization='normalized', k=num_clusters+extra_dim)
190            norms = np.sum(vec*vec,axis=1)
191            T = sparse.spdiags(norms**(-1/2),0,n,n)
192            vec = T@vec  #Normalize rows
193        else:
194            sys.exit("Invalid spectral clustering method " + method)
195
196        kmeans = cluster.KMeans(n_clusters=num_clusters).fit(vec)
197
198        return kmeans.labels_

200class fokker_planck(clustering):
201    def __init__(self, W, num_clusters, beta=0.5, t=1, rho=None):
202        """FokkerPlanck clustering
203        ===================
204
205        Implements the Fokker-Planck clustering algorithm from [1].
206
207        Parameters
208        ----------
209        W : numpy array, scipy sparse matrix, or graphlearning graph object
210            Weight matrix representing the graph.
211        num_clusters : int
212            Number of desired clusters.
213        beta : float (optional), default=0.5
214            Interpolation parameter between mean shift and diffusion.
215        t : float (optional), default=1
216            Time to run Fokker-Planck equation
217        rho : numpy array (optional), default=None
218            Density estimator for mean shift. Default is uniform density.
219
220        Examples
221        ----
222        Fokker-Planck clustering on the two-skies dataset: [fokker_planck_clustering.py](https://github.com/jwcalder/GraphLearning/blob/master/examples/fokker_planck_clustering.py).
223        ```py
224        import numpy as np
225        import graphlearning as gl
226        import matplotlib.pyplot as plt
227
228        X,L = gl.datasets.two_skies(1000)
229        W = gl.weightmatrix.knn(X,10)
230
231        knn_ind,knn_dist = gl.weightmatrix.knnsearch(X,50)
232        rho = 1/np.max(knn_dist,axis=1)
233
234        model = gl.clustering.fokker_planck(W,num_clusters=2,t=1000,beta=0.5,rho=rho)
235        labels = model.fit_predict()
236
237        plt.scatter(X[:,0],X[:,1], c=labels)
238        plt.show()
239        ```
240
241        Reference
242        ---------
243        [1] K. Craig, N.G. Trillos, & D. Slepev. (2021). Clustering dynamics on graphs: from spectral clustering to mean shift through Fokker-Planck interpolation. arXiv:2108.08687.
244        """
245        super().__init__(W, num_clusters)
246            
247        self.beta = beta
248        self.t = t
249        if rho is None:
250            self.rho = np.ones(W.shape[0])
251        else:
252            self.rho = rho
253
254    def _fit(self, all_labels=None):
255
256        beta = self.beta
257        t = self.t
258        rhoinv = 1/self.rho
259
260        #Coifman/Lafon
261        Q1 = -self.graph.laplacian(normalization='coifmanlafon')
262
263        #Mean shift transition matrix
264        Qms = self.graph.gradient(rhoinv, weighted=True).T
265        Qms[Qms<0] = 0
266        Qms = Qms - graph.graph(Qms).degree_matrix()
267
268        #Interplation
269        Q = beta*Qms + (1-beta)*Q1
270        Q = Q.toarray()
271
272        #Matrix exponential
273        #expQt = sparse.linalg.expm(Q*t)
274        #Y = expQt.toarray()
275        expQt = linalg.expm(Q*t)
276
277        #kmeans
278        kmeans = cluster.KMeans(n_clusters=self.num_clusters).fit(expQt)
279
280        return kmeans.labels_

282class incres(clustering):
283    def __init__(self, W, num_clusters, speed=5, T=200):
284        """INCRES clustering
285        ===================
286
287        Implements the INCRES clustering algorithm from [1].
288
289        Parameters
290        ----------
291        W : numpy array, scipy sparse matrix, or graphlearning graph object
292            Weight matrix representing the graph.
293        num_clusters : int
294            Number of desired clusters.
295        speed : float (optional), default=5
296            Speed parameter.
297        T : int (optional), default=100
298            Number of iterations.
299
300        Example
301        ----
302        INCRES clustering on MNIST: [incres_mnist.py](https://github.com/jwcalder/GraphLearning/blob/master/examples/incres_mnist.py).
303        ```py
304        import graphlearning as gl
305
306        W = gl.weightmatrix.knn('mnist', 10, metric='vae')
307        labels = gl.datasets.load('mnist', labels_only=True)
308
309        model = gl.clustering.incres(W, num_clusters=10)
310        pred_labels = model.fit_predict(all_labels=labels)
311        
312        accuracy = gl.clustering.clustering_accuracy(pred_labels,labels)
313        print('Clustering Accuracy: %.2f%%'%accuracy)
314        ```
315
316        Reference
317        ---------
318        [1] X. Bresson, H. Hu, T. Laurent, A. Szlam, and J. von Brecht. [An incremental reseeding strategy for clustering](https://arxiv.org/pdf/1406.3837.pdf). In International Conference on Imaging, Vision and Learning based on Optimization and PDEs (pp. 203-219), 2016.
319        """
320        super().__init__(W, num_clusters)
321            
322        self.speed = speed
323        self.T = T
324
325    def _fit(self, all_labels=None):
326
327        #Short cuts
328        n = self.graph.num_nodes
329        speed = self.speed
330        T = self.T
331        k = self.num_clusters
332
333        #Increment
334        Dm = np.maximum(int(speed*1e-4*n/k),1)
335        
336        #Random initial labeling
337        u = np.random.randint(0,k,size=n)
338
339        #Initialization
340        F = np.zeros((n,k))
341        J = np.arange(n).astype(int)
342
343        #Random walk transition
344        D = self.graph.degree_matrix(p=-1)
345        P = self.graph.weight_matrix*D
346
347        m = int(1)
348        for i in range(T):
349            #Plant
350            F.fill(0)
351            for r in range(k):
352                I = u == r
353                ind = J[I]
354                F[ind[np.random.choice(np.sum(I),m)],r] = 1
355            
356            #Grow
357            while np.min(F) == 0:
358                F = P*F
359
360            #Harvest
361            u = np.argmax(F,axis=1)
362
363            #Increment
364            m = m + Dm
365                
366            #Compute accuracy
367            if all_labels is not None: 
368                acc = clustering_accuracy(u,all_labels)
369                print("Iteration "+str(i)+": Accuracy = %.2f" % acc+"%%, #seeds= %d" % m)
370
371        return u

373def withinss(x):
374    """WithinSS
375    ======
376
377    Clustering of 1D data with WithinSS. Gives exact solution to the 2-means clustering problem
378
379    Parameters
380    ----------
381    x : numpy array
382        1D array of data to cluter.
383
384    Returns
385    -------
386    w : float
387        WithinSS value, essentially the 2-means energy.
388    m : float
389        Threshold that clusters the data array x optimally.
390    """
391
392    x = np.sort(x)
393    n = x.shape[0]
394    sigma = np.std(x)
395    v = np.zeros(n-1,)
396
397    #Initial values for m1,m2
398    x1 = x[:1]
399    x2 = x[1:]
400    m1 = np.mean(x1)
401    m2 = np.mean(x2)
402    for i in range(n-1):
403        v[i] = (i+1)*m1**2 + (n-i-1)*m2**2
404        if i < n-2:
405            m1 = ((i+1)*m1 + x[i+1])/(i+2)
406            m2 = ((n-i-1)*m2 - x[i+1])/(n-i-2)
407    ind = np.argmax(v)
408    m = x[ind]
409    w = (np.sum(x**2) - v[ind])/(n*sigma**2)
410    return w,m

412def RP1D(X,T=100):
413    """Random Projection Clustering
414    ======
415
416    Binary clustering of 1D data with the Random Projection 1D (RP1D) clustering method from [1].
417
418    Parameters
419    ----------
420    X : numpy array
421        (n,d) dimensional array of n datapoints in dimension d.
422    T : int (optional), default=100
423        Number of random projections to try.
424
425    Example
426    -------
427    RP1D clustering on MNIST: [RP1D_mnist.py](https://github.com/jwcalder/GraphLearning/blob/master/examples/RP1D_mnist.py).
428    ```py
429    import graphlearning as gl
430
431    data, labels = gl.datasets.load('mnist')
432
433    x = data[labels <= 1] 
434    y = labels[labels <= 1]
435    y_pred = gl.clustering.RP1D(x,20)
436
437    accuracy = gl.clustering.clustering_accuracy(y_pred, y)
438    print('Clustering Accuracy: %.2f%%'%accuracy)
439    ```
440
441    Returns
442    -------
443    cluster_labels : int
444        0/1 array indicating cluster membership
445
446    References
447    ----------
448    [1] S. Han and M. Boutin. [The hidden structure of image datasets.](https://ieeexplore.ieee.org/stamp/stamp.jsp?arnumber=7350969&casa_token=UsN9y0textMAAAAA:-K9r-Sv4njFQ_txJUpkqCbavM-wTA2CmkgU3co7RjmjTKdcP3guTjahyHA7jZBs1WZTz-E2fETQ&tag=1) 2015 IEEE International Conference on Image Processing (ICIP). IEEE, 2015.
449    """
450
451    n = X.shape[0]
452    d = X.shape[1]
453    v = np.random.rand(T,d)
454    wmin = np.inf
455    imin = 0;
456    for i in range(T):
457        x = np.sum(v[i,:]*X,axis=1)
458        w,m = withinss(x)
459        if w < wmin:
460            wmin = w
461            imin = i
462    x = np.sum(v[imin,:]*X,axis=1)
463    w,m = withinss(x)
464
465    cluster_labels = np.zeros(n,)
466    cluster_labels[x>m] = 1
467
468    return cluster_labels

470def clustering_accuracy(pred_labels,true_labels):
471    """Clustering accuracy
472    ======
473
474    Accuracy for clustering in graph learning. Uses a linear sum assignment
475    to find the best permutation of cluster labels.
476
477    Parameters
478    ----------
479    pred_labels : numpy array, int
480        Predicted labels
481    true_labels : numpy array, int
482        True labels
483
484    Returns
485    -------
486    accuracy : float
487        Accuracy as a number in [0,100].
488    """
489
490    unique_classes = np.unique(true_labels)
491    num_classes = len(unique_classes)
492
493    C = np.zeros((num_classes, num_classes), dtype=float)
494    for i in range(num_classes):
495        for j in range(num_classes):
496            C[i][j] = np.sum((pred_labels == i) & (true_labels != j))
497    row_ind, col_ind = opt.linear_sum_assignment(C)
498
499    return 100*(1-C[row_ind,col_ind].sum()/len(pred_labels))

502def purity(cluster_labels,true_labels):
503    """Clustering purity
504    ======
505
506    Computes clustering purity, which is the fraction of nodes
507    that belong to the largest class in each cluster.
508
509    Parameters
510    ----------
511    cluster_labels : numpy array, int
512        Cluster labels
513    true_labels : numpy array, int
514        True labels
515
516    Returns
517    -------
518    purity : float
519        Purity as a number in [0,100].
520    """
521
522    classes = np.unique(true_labels)
523    num_classes = len(classes)
524    clusters = np.unique(cluster_labels)
525    num_clusters = len(clusters)
526
527    purity = []
528    size = []
529    for c in clusters:
530        labels = true_labels[cluster_labels == c]
531        purity += [np.max(np.bincount(labels))]
532        size += [len(labels)]
533    purity = np.array(purity)
534    size = np.array(size)
535
536    return 100*np.sum(purity)/np.sum(size), purity/size

374def withinss(x):
375    """WithinSS
376    ======
377
378    Clustering of 1D data with WithinSS. Gives exact solution to the 2-means clustering problem
379
380    Parameters
381    ----------
382    x : numpy array
383        1D array of data to cluter.
384
385    Returns
386    -------
387    w : float
388        WithinSS value, essentially the 2-means energy.
389    m : float
390        Threshold that clusters the data array x optimally.
391    """
392
393    x = np.sort(x)
394    n = x.shape[0]
395    sigma = np.std(x)
396    v = np.zeros(n-1,)
397
398    #Initial values for m1,m2
399    x1 = x[:1]
400    x2 = x[1:]
401    m1 = np.mean(x1)
402    m2 = np.mean(x2)
403    for i in range(n-1):
404        v[i] = (i+1)*m1**2 + (n-i-1)*m2**2
405        if i < n-2:
406            m1 = ((i+1)*m1 + x[i+1])/(i+2)
407            m2 = ((n-i-1)*m2 - x[i+1])/(n-i-2)
408    ind = np.argmax(v)
409    m = x[ind]
410    w = (np.sum(x**2) - v[ind])/(n*sigma**2)
411    return w,m
