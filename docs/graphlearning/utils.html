  18def boundary_statistic(X, r, knn=False, return_normals=False, second_order=True, cutoff=True, knn_data=None):
  19    """Boundary test statistic
  20    ===================
  21
  22    Computes the boundary test statistics from [1] for identifying the boundary of a point cloud.
  23
  24    Parameters
  25    ----------
  26    X : (n,d) numpy array (float)
  27        Point cloud in dimension d.
  28    r : float or int,
  29        Radius for test (or numgber of neighbors if knn=True)
  30    knn : bool (optional), default=False
  31        Whether to ues the k-nearest neighbor version of the test, or the radius search version.
  32    return_normals : bool (optional), default=False
  33        Wehther to return estimated normal vectors as well.
  34    second_order : bool (optional), default=True
  35        Whether to use the second order version of the test.
  36    cutoff : bool (optional), default=True
  37        Whether to use the cutoff for the second order test.
  38    knn_data : tuple (optional), default=None
  39        Output of `weightmatrix.knnsearch`, which can be provided to accelerate the computation.
  40
  41    Returns
  42    -------
  43    T : numpy array
  44        Test statistic as a length n numpy array.
  45    nu : (n,d) numpy array
  46        Estimated normals, if `return_normals=True`.
  47
  48    References
  49    ---------
  50    [1] J. Calder, S. Park, and D. Slepčev. [Boundary Estimation from Point Clouds: Algorithms, Guarantees and Applications.](https://arxiv.org/abs/2111.03217) arXiv:2111.03217, 2021.
  51    """
  52
  53    #Estimation of normal vectors
  54    n = X.shape[0]
  55    d = X.shape[1]
  56    if knn:
  57        k = r
  58        #Run knnsearch only if knn_data is not provided
  59        if knn_data is None:
  60            J,D = weightmatrix.knnsearch(X,k)
  61        else:
  62            J,D = knn_data
  63        W = weightmatrix.knn(X, k, kernel='uniform', symmetrize=False, knn_data=(J,D))
  64    else:
  65        W = weightmatrix.epsilon_ball(X, r, kernel='uniform')
  66        
  67    deg = W*np.ones(n)
  68    if np.min(deg)==1:
  69        print('\nWarning: Some points have no neighbors!!!\n')
  70
  71    #Estimation of normals
  72    if second_order:
  73        if knn:
  74            theta = graph.graph(W).degree_matrix(p=-1)
  75        else:
  76            W2 = weightmatrix.epsilon_ball(X, r/2, kernel='uniform')
  77            theta = graph.graph(W).degree_matrix(p=-1)
  78        nu = -graph.graph(W*theta).laplacian()*X
  79    else:
  80        nu = -graph.graph(W).laplacian()*X
  81
  82    #Normalize to unit norm
  83    norms = np.sqrt(np.sum(nu*nu,axis=1))
  84    nu = nu/norms[:,np.newaxis]
  85
  86    #Switch to knn if not selected
  87    if not knn:
  88        k = int(np.max(W*np.ones(W.shape[0]))) #Number of neighbors to use in knnsearch
  89        J,D = weightmatrix.knnsearch(X,k); J=J[:,1:]; D=D[:,1:] #knnsearch and remove self
  90
  91    #Difference between center point and neighbors
  92    V = X[:,np.newaxis,:] - X[J] #(x^0-x^i), nxkxd array
  93
  94    #Compute boundary statistic to all neighbors
  95    if second_order:
  96        nu2 = (nu[:,np.newaxis,:] + nu[J])/2
  97        if cutoff:
  98            nn_mask = np.sum(nu[:,np.newaxis,:]*nu[J],axis=2) > 0
  99            nn_mask = nn_mask[:,:,np.newaxis]
 100            nu2 = nn_mask*nu2 + (1-nn_mask)*nu[:,np.newaxis,:]
 101        xd = np.sum(V*nu2,axis=2) #xd coordinate (nxk)
 102    else: #First order boundary test 
 103        xd = np.sum(V*nu[:,np.newaxis,:],axis=2) #xd coordinate (nxk)
 104
 105    #Return test statistic, masking out to B(x,r), and normals if return_normals=True
 106    if knn:
 107        T = np.max(xd,axis=1)
 108    else:
 109        T = np.max(xd*(D<=r),axis=1)
 110
 111    if return_normals:
 112        return T,nu
 113    else:
 114        return T

 263def sparse_max(A,B):
 264    """Max of two sparse matrices
 265    ======
 266
 267    Computes the elementwise max of two sparse matrices.
 268    Matrices should both be nonegative and square.
 269
 270    Parameters
 271    ----------
 272    A : (n,n) scipy sparse matrix
 273        First matrix.
 274    B : (n,n) scipy sparse matrix
 275        Second matrix.
 276
 277    Returns
 278    -------
 279    C : (n,n) scipy sparse matrix
 280        Sparse max of A and B
 281    """
 282
 283    I = (A + B) > 0
 284    IB = B>A
 285    IA = I - IB
 286    return A.multiply(IA) + B.multiply(IB)

 321def constrained_solve(L,I,g,f=None,x0=None,tol=1e-10):
 322    """Constrained Solve
 323    ======
 324
 325    Uses preconditioned [Conjugate Gradient Method](https://en.wikipedia.org/wiki/Conjugate_gradient_method) 
 326    to solve the equation \\(Lx=f\\) subject to \\(x=g\\) on a contraint set. \\(L\\) must be positive
 327    definite and symmetric.
 328
 329    Parameters
 330    ----------
 331    L : (n,n) numpy array or scipy sparse matrix
 332        Left hand side of linear equation.
 333    I : numpy array (bool or int)
 334        Indices of contraint set.
 335    g : numpy array (float)
 336        Constrained values
 337    f : numpy array (optional), default=None
 338        Right hand side of linear equation. Default is interpreted as \\(f=0\\).
 339    x0 : numpy array (optional), default=None
 340        Initial condition. Default is zero.
 341    tol : float (optional), default = 1e-10
 342        Tolerance for the conjugate gradient method.
 343
 344    Returns
 345    -------
 346    x : numpy array
 347        Solution of linear equation with constraints.
 348    """
 349
 350    L = L.tocsr()
 351    n = L.shape[0]
 352
 353    #Locations of labels
 354    idx = np.full((n,), True, dtype=bool)
 355    idx[I] = False
 356
 357    #Right hand side
 358    b = -L[:,I]*g
 359    b = b[idx]
 360
 361    if f is not None:
 362        b = b + f[idx]
 363
 364    #Left hand side matrix
 365    A = L[idx,:]
 366    A = A[:,idx]
 367    
 368
 369    #Conjugate gradient with Jacobi preconditioner
 370    m = A.shape[0]
 371    M = A.diagonal()
 372    M = sparse.spdiags(1/(M+1e-10),0,m,m).tocsr()
 373
 374    if x0 is None:
 375        v,i = sparse.linalg.cg(A,b,tol=tol,M=M)
 376    else:
 377        v,i = sparse.linalg.cg(A,b,x0=x0[idx],tol=tol,M=M)
 378
 379    #Add labels back into array
 380    u = np.ones((n,))
 381    u[idx] = v
 382    u[I] = g
 383
 384    return u

 386def dirichlet_eigenvectors(L,ind,k):
 387    """Dirichlet eigenvectors
 388    ======
 389
 390    Finds the smallest magnitude Dirichlet eigenvectors/eigenvalues of a symmetric matrix \\(L\\), which satisfy 
 391    \\(x_i=0\\) for \\(i\\in \\Gamma\\) and \\(Lx_i=\\lambda x_i\\) for \\(i\\not\\in \\Gamma\\).
 392
 393    Parameters
 394    ----------
 395    L : (n,n) numpy array or scipy sparse matrix
 396        Matrix to compute eigenvectors of.
 397    ind : numpy array (bool or int)
 398        Indices or boolean mask indicating contraint set \\(\\Gamma\\).
 399    k : int 
 400        Number of eigenvectors to find.
 401
 402    Returns
 403    -------
 404    vals : numpy array
 405        Eigenvalues in increasing order.
 406    vecs : numpy array
 407        Corresponding eigenvectors as columns.
 408    """
 409
 410
 411    L = L.tocsr()
 412    n = L.shape[0]
 413
 414    #Locations of labels
 415    idx = np.full((n,), True, dtype=bool)
 416    idx[ind] = False
 417
 418    #Left hand side matrix
 419    A = L[idx,:]
 420    A = A[:,idx]
 421    
 422    #Eigenvector solver
 423    vals, vec = sparse.linalg.eigsh(A,k=k,which='SM')
 424    
 425    #Add labels back into array
 426    vecs = np.zeros((n,k))
 427    vecs[idx,:] = vec
 428
 429    if k == 1:
 430        vecs = vecs.flatten()
 431
 432    return vals, vecs

 435def constrained_solve_gmres(L,f,R,g,ind,tol=1e-5):
 436    """Constrained GMRES Solve
 437    ======
 438
 439    Uses preconditioned [GMRES](https://en.wikipedia.org/wiki/Generalized_minimal_residual_method) to solve
 440    the equation \\(Lx=f\\) subject to \\(Rx=g\\) on a contraint set.
 441
 442    Parameters
 443    ----------
 444    L : (n,n) numpy array or scipy sparse matrix
 445        Left hand side of linear equation.
 446    f : (n,1) numpy array
 447        Right hand side of linear equation.
 448    R : (n,n) numpy array or scipy sparse matrix
 449        Constraint matrix.
 450    g : numpy array
 451        Length n numpy array for boundary constriants.
 452    ind : numpy array (bool or int)
 453        Indices or boolean mask indicating contraint set.
 454    tol : float (optional), default = 1e-5
 455        Tolerance for GMRES.
 456
 457    Returns
 458    -------
 459    x : numpy array
 460        Solution of linear equation with constraints.
 461    """
 462
 463    #Mix matrices based on boundary points
 464    A = L.copy()
 465    A = A.tolil()
 466    A[ind,:] = R[ind,:]
 467    A = A.tocsr()
 468
 469    #Right hand side
 470    b = f.copy()
 471    b[ind] = g[ind]
 472
 473    #Preconditioner
 474    m = A.shape[0]
 475    M = A.diagonal()
 476    M = sparse.spdiags(1/M,0,m,m).tocsr()
 477
 478    #GMRES solver
 479    u,info = sparse.linalg.gmres(A, b, M=M, tol=tol)
 480
 481    return u

 483def conjgrad(A, b, x0=None, max_iter=1e5, tol=1e-10):
 484    r"""Conjugate Gradient Method
 485    ======
 486
 487    Conjugate gradient method for solving the linear equation
 488    \\[Ax = b\\]
 489    where \\(A\\in \\mathbb{R}^{n\\times n}\\) is symmetric and positive semi-definite, 
 490    \\(x\\in \\mathbb{R}^{n\\times k}\\) and \\(b\\in \\mathbb{R}^{n\\times k}\\).
 491
 492    Parameters
 493    ----------
 494    A : (n,n) numpy array or scipy sparse matrix
 495        Left hand side of linear equation.
 496    b : (n,k) numpy array
 497        Right hand side of linear equation.
 498    x0 : (n,k) numpy array (optional)
 499        Initial guess. If not provided, then x0=0.
 500    max_iter : int (optional), default = 1e5
 501        Maximum number of iterations.
 502    tol : float (optional), default = 1e-10
 503        Tolerance for stopping conjugate gradient iterations.
 504
 505    Returns
 506    -------
 507    x : (n,k) numpy array
 508        Solution of \\(Ax=b\\) with conjugate gradient
 509    """
 510
 511    if x0 is None:
 512        x = np.zeros_like(b)
 513    else:
 514        x = x0
 515
 516    r = b - A@x
 517    p = r
 518    rsold = np.sum(r**2,axis=0)
 519  
 520    err = 1 
 521    i = 0
 522    while (err > tol) and (i < max_iter):
 523        i += 1
 524        Ap = A@p
 525        alpha = rsold / np.sum(p*Ap,axis=0)
 526        x += alpha * p
 527        r -= alpha * Ap
 528        rsnew = np.sum(r**2,axis=0)
 529        err = np.sqrt(np.sum(rsnew)) 
 530        p = r + (rsnew / rsold) * p
 531        rsold = rsnew
 532
 533    return x

 624def randomized_svd(A, k=10, c=None, q=1):
 625    """Randomized SVD
 626    ======
 627
 628    Approximates top k singular values and vectors of A with a randomized
 629    SVD algorithm.
 630
 631   
 632    Parameters
 633    ----------
 634    A : numpy array or matrix, scipy sparse matrix, or sparse linear operator
 635        Matrix to compute SVD of.
 636    k : int (optional), default=10
 637        Number of eigenvectors to compute.
 638    q : int (optional), default=1
 639        Exponent to use in randomized svd.
 640    c : int (optional), default=2*k
 641        Cutoff for randomized SVD.
 642
 643    Returns
 644    -------
 645    u : (n,k) numpy array, float 
 646        Unitary matrix having left singular vectors as columns. 
 647    s : numpy array, float
 648        The singular values.
 649    vt : (k,n) numpy array, float
 650        Unitary matrix having right singular vectors as rows.
 651
 652    Reference
 653    ---------
 654    Halko, Nathan, Per-Gunnar Martinsson, and Joel A. Tropp. [Finding structure 
 655    with randomness: Probabilistic algorithms for constructing approximate matrix 
 656    decompositions.](https://arxiv.org/abs/0909.4061) SIAM review 53.2 (2011): 217-288.
 657    """
 658
 659
 660    if c is None:
 661        c = 2*k
 662
 663    n = A.shape[1]
 664
 665    #Random Gaussian projection
 666    Omega = np.random.randn(n,c)
 667    Y = A@Omega
 668    for i in range(q):
 669        Y = A@(A.T@Y)
 670
 671    #QR Factorization
 672    Q,R = np.linalg.qr(Y)
 673
 674    #SVD
 675    B = Q.T@A
 676    u,s,vt = linalg.svd(B, full_matrices=False)
 677    u = Q@u
 678
 679    #Sort singular values from largest to smallest
 680    ind = np.argsort(-s)
 681    u = u[:,ind]
 682    s = s[ind]
 683    vt = vt[ind,:]
 684
 685    #Truncate to k
 686    u = u[:,:k]
 687    s = s[:k]
 688    vt = vt[:k,:]
 689
 690    return u,s,vt

 693def rand_annulus(n,d,r1,r2):
 694    """Random points in annulus
 695    ======
 696
 697    Generates independent and uniformly distributed random variables in the annulus
 698    \\(B_{r_2} \\setminus B_{r_1}\\).
 699   
 700    Parameters
 701    ----------
 702    n : int 
 703        Number of points.
 704    d : int
 705        Dimension.
 706    r1 : float
 707        Inner radius.
 708    r2 : float
 709        Outer radius
 710
 711    Returns
 712    -------
 713    X : (n,d) numpy array
 714        Random points in annulus.
 715    """
 716
 717    N = 0
 718    X = np.zeros((1,d))
 719    while X.shape[0] <= n:
 720
 721        Y = r2*(2*np.random.rand(n,d) - 1)
 722        dist2 = np.sum(Y*Y,axis=1) 
 723        I = (dist2 < r2*r2)&(dist2 > r1*r1)
 724        Y = Y[I,:]
 725        X = np.vstack((X,Y))
