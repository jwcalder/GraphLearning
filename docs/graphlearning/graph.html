  25        ========

 105    def degree_matrix(self, p=1):
 106        """Degree Matrix
 107        ======
 108
 109        Given a weight matrix \\(W\\), returns the diagonal degree matrix 
 110        in the form
 111        \\[D_{ii} = \\left(\\sum_{j=1}^n w_{ij}\\right)^p.\\]
 112
 113        Parameters
 114        ----------
 115        p : float (optional), default=1
 116            Optional exponent to apply to the degree.
 117
 118        Returns
 119        -------
 120        D : (n,n) scipy sparse matrix, float
 121            Sparse diagonal degree matrix.
 122        """
 123
 124        #Construct sparse degree matrix
 125        d = self.degree_vector()
 126        D = sparse.spdiags(d**p, 0, self.num_nodes, self.num_nodes)

 128        return D.tocsr()

 187    def gradient(self, u, weighted=False, p=0.0):
 188        """Graph Gradient
 189        ======
 190
 191        Computes the graph gradient \\(\\nabla u\\) of \\(u\\in \\mathbb{R}^n\\), which is
 192        the sparse matrix with the form
 193        \\[\\nabla u_{ij} = u_j - u_i,\\]
 194        whenever \\(w_{ij}>0\\), and \\(\\nabla u_{ij}=0\\) otherwise.
 195        If `weighted=True` is chosen, then the gradient is weighted by the graph weight 
 196        matrix as follows
 197        \\[\\nabla u_{ij} = w_{ij}^p(u_j - u_i).\\]
 198
 199        Parameters
 200        ----------
 201        u : numpy array, float
 202            Vector (graph function) to take gradient of
 203        weighted : bool (optional), default=False,True
 204            Whether to weight the gradient by the graph weight matrix. Default is False when p=0 and True when \\(p\\neq 0\\).
 205        p : float (optional), default=0,1
 206            Power for weights on weighted gradient. Default is 0 when unweighted and 1 when weighted.
 207
 208        Returns
 209        -------
 210        G : (n,n) scipy sparse matrix, float
 211            Sparse graph gradient matrix
 212        """
 213
 214        n = self.num_nodes
 215
 216        if p != 0.0:
 217            weighted = True
 218
 219        if weighted == True and p==0.0:
 220            p = 1.0
 221
 222        if weighted:
 223            G = sparse.coo_matrix(((self.V**p)*(u[self.J]-u[self.I]), (self.I,self.J)),shape=(n,n)).tocsr()
 224        else:
 225            G = sparse.coo_matrix((u[self.J]-u[self.I], (self.I,self.J)),shape=(n,n)).tocsr()
 226
 227        return G

 229    def divergence(self, V, weighted=True):
 230        """Graph Divergence
 231        ======
 232
 233        Computes the graph divergence \\(\\text{div} V\\) of a vector field \\(V\\in \\mathbb{R}^{n\\times n}\\), 
 234        which is the vector 
 235        \\[\\nabla u_{ij} = u_j - u_i,\\]
 236        If `weighted=True` is chosen, then the divergence is weighted by the graph weight 
 237        matrix as follows
 238        \\[\\nabla u_{ij} = w_{ij}(u_j - u_i).\\]
 239
 240        Parameters
 241        ----------
 242        V : scipy sparse matrix, float
 243            Sparse matrix representing a vector field over the graph.
 244        weighted : bool (optional), default=True
 245            Whether to weight the divergence by the graph weight matrix.
 246
 247        Returns
 248        -------
 249        divV : numpy array
 250            Divergence of V.
 251        """
 252    
 253        V = V - V.transpose()
 254
 255        if weighted:
 256            V = V.multiply(self.weight_matrix)
 257
 258        divV = V*np.ones(self.num_nodes)/2
 259
 260        return divV

 264        """Reweight a weight matrix
 265        ======
 266
 267        Reweights the graph weight matrix more heavily near labeled nodes. Used in semi-supervised
 268        learning at very low label rates. [Need to describe all methods...]

 271        ----------
 272        idx : numpy array (int)

 277        normalization : {'combinatorial','normalized'}, default='combinatorial'

 305        """

 307        if method == 'poisson':
 308            
 309            n = self.num_nodes
 310            f = np.zeros(n)
 311            f[idx] = 1
 312
 313            if normalization == 'combinatorial':
 314                f -= np.mean(f)
 315                L = self.laplacian()
 316            elif normalization == 'normalized':
 317                d = self.degree_vector()**(0.5)
 318                c = np.sum(d*f)/np.sum(d)
 319                f -= c
 320                L = self.laplacian(normalization=normalization)
 321            else:
 322                sys.exit('Unsupported normalization '+normalization+' for graph.reweight.')

 325            w -= np.min(w)
 326            w += 1e-5
 327            D = sparse.spdiags(w,0,n,n).tocsr()

 364    def laplacian(self, normalization="combinatorial", alpha=1):
 365        """Graph Laplacian
 366        ======
 367
 368        Computes various normalizations of the graph Laplacian for a 
 369        given weight matrix \\(W\\). The choices are
 370        \\[L_{\\rm combinatorial} = D - W,\\]
 371        \\[L_{\\rm randomwalk} = I - D^{-1}W,\\]
 372        and
 373        \\[L_{\\rm normalized} = I - D^{-1/2}WD^{-1/2},\\]
 374        where \\(D\\) is the diagonal degree matrix, which is defined as
 375        \\[D_{ii} = \\sum_{j=1}^n w_{ij}.\\]
 376        The Coifman-Lafon Laplacian is also supported. 
 377
 378        Parameters
 379        ----------
 380        normalization : {'combinatorial','randomwalk','normalized','coifmanlafon'}, default='combinatorial'
 381            Type of normalization to apply.
 382        alpha : float (optional)
 383            Parameter for Coifman-Lafon Laplacian
 384
 385        Returns
 386        -------
 387        L : (n,n) scipy sparse matrix, float
 388            Graph Laplacian as sparse scipy matrix.
 389        """
 390
 391        I = sparse.identity(self.num_nodes)
 392        D = self.degree_matrix()
 393
 394        if normalization == "combinatorial":
 395            L = D - self.weight_matrix
 396        elif normalization == "randomwalk":
 397            Dinv = self.degree_matrix(p=-1)
 398            L = I - Dinv*self.weight_matrix
 399        elif normalization == "normalized":
 400            Dinv2 = self.degree_matrix(p=-0.5)
 401            L = I - Dinv2*self.weight_matrix*Dinv2
 402        elif normalization == "coifmanlafon":
 403            D = self.degree_matrix(p=-alpha)
 404            L = graph(D*self.weight_matrix*D).laplacian(normalization='randomwalk')
 405        else:
 406            sys.exit("Invalid option for graph Laplacian normalization.")
 407
 408        return L.tocsr()

 410    def infinity_laplacian(self,u):
 411        """Graph Infinity Laplacian
 412        ======
 413
 414        Computes the graph infinity Laplacian of a vector \\(u\\), given by
 415        \\[L_\\infty u_i= \\min_j w_{ij}(u_j-u_i) + \\max_j w_{ij} (u_j-u_i).\\]
 416               
 417        Returns
 418        -------
 419        Lu : numpy array
 420            Graph infinity Laplacian.
 421        """
 422
 423        n = self.num_nodes
 424        M = sparse.coo_matrix((self.V*(u[self.J]-u[self.I]), (self.I,self.J)),shape=(n,n)).tocsr()
 425        M = M.min(axis=1) + M.max(axis=1)
 426        Lu = M.toarray().flatten()
 427
 428        return Lu

 480    def eigen_decomp(self, normalization='combinatorial', method='exact', k=10, c=None, gamma=0, tol=0, q=1):
 481        """Eigen Decomposition of Graph Laplacian
 482        ======
 483
 484        Computes the the low-lying eigenvectors and eigenvalues of 
 485        various normalizations of the graph Laplacian. Computations can 
 486        be either exact, or use a fast low-rank approximation via 
 487        randomized SVD. 
 488
 489        Parameters
 490        ----------
 491        normalization : {'combinatorial','randomwalk','normalized'}, default='combinatorial'
 492            Type of normalization of graph Laplacian to apply.
 493        method : {'exact','lowrank'}, default='exact'
 494            Method for computing eigenvectors. 'exact' uses scipy.sparse.linalg.eigs, while
 495            'lowrank' uses a low rank approximation via randomized SVD. Lowrank is not 
 496            implemented for gamma > 0.
 497        k : int (optional), default=10
 498            Number of eigenvectors to compute.
 499        c : int (optional), default=2*k
 500            Cutoff for randomized SVD.
 501        gamma : float (optional), default=0
 502            Parameter for modularity (add more details)
 503        tol : float (optional), default=0
 504            tolerance for eigensolvers.
 505        q : int (optional), default=1
 506            Exponent to use in randomized svd.
 507
 508        Returns
 509        -------
 510        vals : numpy array, float 
 511            eigenvalues in increasing order.
 512        vecs : (n,k) numpy array, float
 513            eigenvectors as columns.
 514
 515        Example
 516        -------
 517        This example compares the exact and lowrank (ranomized svd) methods for computing the spectrum: [randomized_svd.py](https://github.com/jwcalder/GraphLearning/blob/master/examples/randomized_svd.py).
 518        ```py
 519        import numpy as np
 520        import matplotlib.pyplot as plt
 521        import sklearn.datasets as datasets
 522        import graphlearning as gl
 523
 524        X,L = datasets.make_moons(n_samples=500,noise=0.1)
 525        W = gl.weightmatrix.knn(X,10)
 526        G = gl.graph(W)
 527
 528        num_eig = 7
 529        vals_exact, vecs_exact = G.eigen_decomp(normalization='normalized', k=num_eig, method='exact')
 530        vals_rsvd, vecs_rsvd = G.eigen_decomp(normalization='normalized', k=num_eig, method='lowrank', q=50, c=50)
 531
 532        for i in range(1,num_eig):
 533            rsvd = vecs_rsvd[:,i]
 534            exact = vecs_exact[:,i]
 535
 536            sign = np.sum(rsvd*exact)
 537            if sign < 0:
 538                rsvd *= -1
 539
 540            err = np.max(np.absolute(rsvd - exact))/max(np.max(np.absolute(rsvd)),np.max(np.absolute(exact)))
 541
 542            fig, (ax1,ax2) = plt.subplots(1,2, figsize=(10,5))
 543            fig.suptitle('Eigenvector %d, err=%f'%(i,err))
 544
 545            ax1.scatter(X[:,0],X[:,1], c=rsvd)
 546            ax1.set_title('Random SVD')
 547
 548            ax2.scatter(X[:,0],X[:,1], c=exact)
 549            ax2.set_title('Exact')
 550
 551        plt.show()
 552        ```

 560        same_k = self.eigendata[normalization]['k'] == k
 561        same_c = self.eigendata[normalization]['c'] == c
 562        same_gamma = self.eigendata[normalization]['gamma'] == gamma
 563        same_tol = self.eigendata[normalization]['tol'] == tol
 564        same_q = self.eigendata[normalization]['q'] == q
 565
 566        #If already computed, then return eigenvectors
 567        if same_method and same_k and same_c and same_gamma and same_tol and same_q:
 568        
 569            return self.eigendata[normalization]['eigenvalues'], self.eigendata[normalization]['eigenvectors']
 570        
 571        #Else, we need to compute the eigenvectors
 572        else:
 573            self.eigendata[normalization]['method'] = method 
 574            self.eigendata[normalization]['k'] = k
 575            self.eigendata[normalization]['c'] = c
 576            self.eigendata[normalization]['gamma'] = gamma
 577            self.eigendata[normalization]['tol'] = tol
 578            self.eigendata[normalization]['q'] = q
 579
 580            n = self.num_nodes
 581
 582            #If not using modularity
 583            if gamma == 0:

 585                if normalization == 'randomwalk' or normalization == 'normalized':
 586
 587                    D = self.degree_matrix(p=-0.5)
 588                    A = D*self.weight_matrix*D
 589
 590                    if method == 'exact':
 591                        u,s,vt = splinalg.svds(A, k=k, tol=tol)
 592                    elif method == 'lowrank':

 643                #If using modularity
 644                deg = self.degree_vector()
 645                deg1 = deg**p1
 646                deg2 = deg**p2
 647                m = np.sum(deg)/2 
 648                def M(v):
 649                    v = v.flatten()
 650                    return (lap*v).flatten() + (gamma/m)*(deg2.T@v)*deg1
 651
 652                L = sparse.linalg.LinearOperator((n,n), matvec=M)
 653                vals, vecs = sparse.linalg.eigsh(L, k=k, which='SM', tol=tol)
 654
 655                #Correct for random walk Laplacian if chosen
 656                vecs = P@vecs
 657
 658
 659            #Store eigenvectors for resuse later
 660            self.eigendata[normalization]['eigenvalues'] = vals
 661            self.eigendata[normalization]['eigenvectors'] = vecs
 662
 663            return vals, vecs

 978    def plaplace(self, bdy_set, bdy_val, p, tol=1e-1, max_num_it=1e6, prog=False, fast=True):
 979        """Game-theoretic p-Laplacian
 980        ======
 981
 982        Computes the solution of the game-theoretic p-Laplace equation \\(L_p u_i=0\\) 
 983        for \\(i\\not\\in \\Gamma\\), subject to \\(u_i=g_i\\) for \\(i\\in \\Gamma\\).
 984        The game-theoretic p-Laplacian is given by
 985        \\[ L_p u = \\frac{1}{p}L_{\\rm randomwalk} + \\left(1-\\frac{2}{p}\\right)L_\\infty u,\\]
 986        where \\(L_{\\rm randomwalk}\\) is the random walk graph Laplacian and \\(L_\\infty\\) is the
 987        graph infinity-Laplace operator, given by
 988        \\[ L_\\infty u_i = \\min_j w_{ij}(u_i-u_j) + \\max_j w_{ij} (u_i-u_j).\\]
 989
 990        Parameters
 991        ----------
 992        bdy_set : numpy array (int or bool) 
 993            Indices or boolean mask indicating the boundary nodes \\(\\Gamma\\).
 994        bdy_val : numpy array or single float (optional), default=0
 995            Boundary values \\(g\\) on \\(\\Gamma\\). A single float is
 996            interpreted as a constant over \\(\\Gamma\\).
 997        p : float
 998            Value of \\(p\\).
 999        tol : float (optional), default=1e-1
1000            Tolerance with which to solve the equation.
1001        max_num_it : int (optional), default=1e6
1002            Maximum number of iterations.
1003        prog : bool (optional), default=False
1004            Toggles whether to print progress information.
1005        fast : bool (optional), default=True
1006            Whether to use constant \\(w_{ij}=1\\) weights for the infinity-Laplacian
1007            which allows a faster algorithm to be used.
1008
1009        Returns
1010        -------
1011        u : numpy array, float 
1012            Solution of graph p-Laplace equation.
1013
1014        Example
1015        -------
1016        This example uses the p-Laplace equation to interpolate boundary values: [plaplace.py](https://github.com/jwcalder/GraphLearning/blob/master/examples/plaplace.py).
1017        ```py
1018        import graphlearning as gl
1019        import numpy as np
1020        import matplotlib.pyplot as plt
1021
1022        X = np.random.rand(int(1e4),2)
1023        x,y = X[:,0],X[:,1]
1024
1025        eps = 0.02
1026        W = gl.weightmatrix.epsilon_ball(X, eps)
1027        G = gl.graph(W)
1028
1029        bdy_set = (x < eps) | (x > 1-eps) | (y < eps) | (y > 1-eps)
1030        bdy_val = x**2 - y**2
1031
1032        u = G.plaplace(bdy_set, bdy_val[bdy_set], p=10)
1033
1034        plt.scatter(x,y,c=u,s=0.25)
1035        plt.scatter(x[bdy_set],y[bdy_set],c='r',s=0.5)
1036        plt.show()
1037        ```
1038        """
1039            
1040        #Import c extensions
1041        from . import cextensions
1042        
1043        n = self.num_nodes
1044        alpha = 1/(p-1)
1045        beta = 1-alpha
1046
1047        #Convert boundary data to standard format
1048        bdy_set, bdy_val = utils._boundary_handling(bdy_set, bdy_val)
1049
1050        #If fast solver
1051        if fast:
1052
1053            u = np.zeros((n,))        #Initial condition
1054
1055            #Type casting and memory blocking
1056            u = np.ascontiguousarray(u,dtype=np.float64)
1057            bdy_set = np.ascontiguousarray(bdy_set,dtype=np.int32)
1058            bdy_val = np.ascontiguousarray(bdy_val,dtype=np.float64)
1059
1060            weighted = False
1061            tol = 1e-6
1062            cextensions.lip_iterate(u,self.J,self.I,self.V,bdy_set,bdy_val,max_num_it,tol,float(prog),float(weighted),float(alpha),float(beta))
1063        else:
1064            uu = np.max(bdy_val)*np.ones((n,))
1065            ul = np.min(bdy_val)*np.ones((n,))
1066
1067            #Set labels
1068            uu[bdy_set] = bdy_val
1069            ul[bdy_set] = bdy_val
1070
1071            #Type casting and memory blocking
1072            uu = np.ascontiguousarray(uu,dtype=np.float64)
1073            ul = np.ascontiguousarray(ul,dtype=np.float64)
1074            bdy_set = np.ascontiguousarray(bdy_set,dtype=np.int32)
1075            bdy_val = np.ascontiguousarray(bdy_val,dtype=np.float64)
1076
1077            cextensions.lp_iterate(uu,ul,self.J,self.I,self.V,bdy_set,bdy_val,p,float(max_num_it),float(tol),float(prog))
1078            u = (uu+ul)/2
1079
1080        return u

1172    def page_rank(self,alpha=0.85,v=None,tol=1e-10):
1173        """PageRank
1174        ======
1175
1176        Solves for the PageRank vector, which is the solution of the PageRank equation
1177        \\[ (I - \\alpha P)u = (1-\\alpha) v, \\]
1178        where \\(P = W^T D^{-1}\\) is the probability transition matrix, with \\(D\\) the diagonal
1179        degree matrix, \\(v\\) is the teleportation distribution, and \\(\\alpha\\) is the 
1180        teleportation paramter. Solution is computed with the power iteration
1181        \\[ u_{k+1} = \\alpha P u_k + (1-\\alpha) v.\\]
1182
1183        Parameters
1184        ----------
1185        alpha : float (optional), default=0.85

 365    def laplacian(self, normalization="combinatorial", alpha=1):
 366        """Graph Laplacian
 367        ======
 368
 369        Computes various normalizations of the graph Laplacian for a 
 370        given weight matrix \\(W\\). The choices are
 371        \\[L_{\\rm combinatorial} = D - W,\\]
 372        \\[L_{\\rm randomwalk} = I - D^{-1}W,\\]
 373        and
 374        \\[L_{\\rm normalized} = I - D^{-1/2}WD^{-1/2},\\]
 375        where \\(D\\) is the diagonal degree matrix, which is defined as
 376        \\[D_{ii} = \\sum_{j=1}^n w_{ij}.\\]
 377        The Coifman-Lafon Laplacian is also supported. 
 378
 379        Parameters
 380        ----------
 381        normalization : {'combinatorial','randomwalk','normalized','coifmanlafon'}, default='combinatorial'
 382            Type of normalization to apply.
 383        alpha : float (optional)
 384            Parameter for Coifman-Lafon Laplacian
 385
 386        Returns
 387        -------
 388        L : (n,n) scipy sparse matrix, float
 389            Graph Laplacian as sparse scipy matrix.
 390        """
 391
 392        I = sparse.identity(self.num_nodes)
 393        D = self.degree_matrix()
 394
 395        if normalization == "combinatorial":
 396            L = D - self.weight_matrix
 397        elif normalization == "randomwalk":
 398            Dinv = self.degree_matrix(p=-1)
 399            L = I - Dinv*self.weight_matrix
 400        elif normalization == "normalized":
 401            Dinv2 = self.degree_matrix(p=-0.5)
 402            L = I - Dinv2*self.weight_matrix*Dinv2
 403        elif normalization == "coifmanlafon":
 404            D = self.degree_matrix(p=-alpha)
 405            L = graph(D*self.weight_matrix*D).laplacian(normalization='randomwalk')
 406        else:
 407            sys.exit("Invalid option for graph Laplacian normalization.")
 408
 409        return L.tocsr()

 980        """Game-theoretic p-Laplacian
 981        ======
 982
 983        Computes the solution of the game-theoretic p-Laplace equation \\(L_p u_i=0\\) 
 984        for \\(i\\not\\in \\Gamma\\), subject to \\(u_i=g_i\\) for \\(i\\in \\Gamma\\).

 986        \\[ L_p u = \\frac{1}{p}L_{\\rm randomwalk} + \\left(1-\\frac{2}{p}\\right)L_\\infty u,\\]
 987        where \\(L_{\\rm randomwalk}\\) is the random walk graph Laplacian and \\(L_\\infty\\) is the
 988        graph infinity-Laplace operator, given by
 989        \\[ L_\\infty u_i = \\min_j w_{ij}(u_i-u_j) + \\max_j w_{ij} (u_i-u_j).\\]
 990
 991        Parameters
 992        ----------
 993        bdy_set : numpy array (int or bool) 
 994            Indices or boolean mask indicating the boundary nodes \\(\\Gamma\\).
 995        bdy_val : numpy array or single float (optional), default=0
 996            Boundary values \\(g\\) on \\(\\Gamma\\). A single float is
 997            interpreted as a constant over \\(\\Gamma\\).
 998        p : float
 999            Value of \\(p\\).
1000        tol : float (optional), default=1e-1
1001            Tolerance with which to solve the equation.
1002        max_num_it : int (optional), default=1e6
1003            Maximum number of iterations.
1004        prog : bool (optional), default=False
1005            Toggles whether to print progress information.
1006        fast : bool (optional), default=True
1007            Whether to use constant \\(w_{ij}=1\\) weights for the infinity-Laplacian
1008            which allows a faster algorithm to be used.
1009
1010        Returns
1011        -------
1012        u : numpy array, float 
1013            Solution of graph p-Laplace equation.
1014
1015        Example
1016        -------
1017        This example uses the p-Laplace equation to interpolate boundary values: [plaplace.py](https://github.com/jwcalder/GraphLearning/blob/master/examples/plaplace.py).

1019        import graphlearning as gl
1020        import numpy as np
1021        import matplotlib.pyplot as plt
1022

1024        x,y = X[:,0],X[:,1]
1025
1026        eps = 0.02
1027        W = gl.weightmatrix.epsilon_ball(X, eps)
1028        G = gl.graph(W)
1029

1031        bdy_val = x**2 - y**2
1032
1033        u = G.plaplace(bdy_set, bdy_val[bdy_set], p=10)
1034

1036        plt.scatter(x[bdy_set],y[bdy_set],c='r',s=0.5)
1037        plt.show()
1038        ```
1039        """
1040            
1041        #Import c extensions
1042        from . import cextensions
1043        
1044        n = self.num_nodes
1045        alpha = 1/(p-1)
1046        beta = 1-alpha
1047
1048        #Convert boundary data to standard format
1049        bdy_set, bdy_val = utils._boundary_handling(bdy_set, bdy_val)
1050
1051        #If fast solver
1052        if fast:
1053
1054            u = np.zeros((n,))        #Initial condition
1055
1056            #Type casting and memory blocking
1057            u = np.ascontiguousarray(u,dtype=np.float64)
1058            bdy_set = np.ascontiguousarray(bdy_set,dtype=np.int32)
1059            bdy_val = np.ascontiguousarray(bdy_val,dtype=np.float64)
1060

1063            cextensions.lip_iterate(u,self.J,self.I,self.V,bdy_set,bdy_val,max_num_it,tol,float(prog),float(weighted),float(alpha),float(beta))

1079            u = (uu+ul)/2

1173    def page_rank(self,alpha=0.85,v=None,tol=1e-10):
1174        """PageRank
1175        ======
1176
1177        Solves for the PageRank vector, which is the solution of the PageRank equation
1178        \\[ (I - \\alpha P)u = (1-\\alpha) v, \\]
1179        where \\(P = W^T D^{-1}\\) is the probability transition matrix, with \\(D\\) the diagonal
1180        degree matrix, \\(v\\) is the teleportation distribution, and \\(\\alpha\\) is the 
1181        teleportation paramter. Solution is computed with the power iteration
1182        \\[ u_{k+1} = \\alpha P u_k + (1-\\alpha) v.\\]
1183
1184        Parameters
1185        ----------
1186        alpha : float (optional), default=0.85
1187            Teleportation parameter.
1188        v : numpy array (optional), default=None
1189            Teleportation distribution. Default is the uniform distribution.
1190        tol : float (optional), default=1e-10
1191            Tolerance with which to solve the PageRank equation.
1192
1193        Returns
1194        -------
1195        u : numpy array, float 
1196            PageRank vector.
1197        """
1198
1199        n = self.num_nodes
1200
1201        u = np.ones((n,))/n
1202        if v is None:
1203            v = np.ones((n,))/n
1204
1205        D = self.degree_matrix(p=-1)
1206        P = self.weight_matrix.T@D

Dijkstra's algorithm

Computes a graph distance function with Dijkstra's algorithm. The graph distance is [ d(x,y) = \min_p \sum_{i=1}^M w_{p_i,p_{i+1}}f_{p_{i+1}},] where the minimum is over paths (p) connecting (x) and (y), (w_{ij}) is the weight from (i) to (j), and (f_i) is an additional per-vertex weights. A path must satisfy (w_{p_i,p_{i+1}}>0) for all (i). Dijkstra's algorithm returns the distance function to a terminal set (\Gamma), given by [u(x) = \min_{i\in \Gamma} {g(x_i) + d(x,x_i)},] where (g) are boundary values. An optional feature also returns the closest point information [cp(x) = \text{argmin}_{i\in \Gamma} {g(x_i) + d(x,x_i)}.] We note that the distance function (u) can also be interpreted as the solution of the graph eikonal equation [ \max_j w_{ji}^{-1} (u(x_i) - u(x_j)) = f_i] subject to (u=g) on (\Gamma).

Parameters
  • bdy_set (numpy array (int)): Indices or boolean mask identifying the boundary nodes (\Gamma).
  • bdy_val (numpy array (float), optional): Boundary values (g) on (\Gamma). A single float is interpreted as a constant over (\Gamma).
  • f (numpy array or scalar float, default=1): Right hand side of eikonal equation. If a scalar, it is extended to a vector over the graph.
  • max_dist (float or np.inf (optional), default = np.inf): Distance at which to terminate Dijkstra's algorithm. Nodes with distance greater than max_dist will contain the value np.inf.
  • return_cp (bool (optional), default=False): Whether to return closest point. Nodes with distance greater than max_dist contain -1 for closest point index.
Returns
  • dist_func (numpy array, float): Distance function computed via Dijkstra's algorithm.
  • cp (numpy array, int): Closest point indices. Only returned if return_cp=True
